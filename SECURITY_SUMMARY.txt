🚀 DEPLOYMENT PROCESS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Step 1: Archive Project
  tar -czf sms-deploy.tar.gz \
    --exclude='.env' \
    --exclude='vendor' \
    --exclude='node_modules' \
    --exclude='database/*.sqlite' \
    sms-management-system/

Step 2: On Production Server
  tar -xzf sms-deploy.tar.gz
  cd sms-management-system
  composer install --no-dev
  cp .env.example .env
  # Edit .env with production credentials
  php artisan migrate --force

Step 3: Verify
  ✅ All code deployed
  ✅ Dependencies installed
  ✅ New .env created with production settings
  ✅ No dev data copied
  ✅ No credentials exposed
